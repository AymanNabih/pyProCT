00010 #     Calculates one of the limits of the cutoff range by adding 'grain' to the starting cutoff until the
00011 #     number of clusters of the produced clustering is into the expected range. The direction can be controlled
00012 #     with the sign of 'grain'.
00013 # 
00014 #     @param starting_cutoff: A guess of the smaller cutoff we can use. It can be 0
00015 #     @param min_clusters: The minimum number of clusters allowed by clustering.
00016 #     @param max_clusters: The maximum number of clusters allowed by clustering.
00017 #     @param grain: The cutoff step that will be used to discover the cutoff range. If it is positive it is going
00018 #     to move to the right, if negative to the left. Exiting conditions are also based on this.
00019 #     @param hie_algorithm: The hierarchical algorithm instance (holds the hie matrix, so that it has not to be
00020 #     calculated each time we get a clustering).
00021 #     
00022 def find_cutoff_limit(starting_cutoff, min_clusters, max_clusters, grain, hie_algorithm):
00023     current_cutoff = starting_cutoff
00024     search_ended = False
00025     while not search_ended :
00026         clustering = hie_algorithm.perform_clustering(kwargs={"cutoff":current_cutoff})
00027         clustering_size = len(clustering.clusters)
00028         # Stop when it is into the allowed range
00029         im_in_the_range = clustering_size >= min_clusters and clustering_size <= max_clusters
00030 
00031         im_out_of_range = False
00032         if grain>0: # we are moving to the 'right', so to bigger cutoffs and smaller clusters
00033             im_out_of_range = clustering_size <= min_clusters
00034         else:
00035             im_out_of_range = clustering_size >= max_clusters
00036 
00037         search_ended =  im_in_the_range or im_out_of_range
00038         #print "current", current_cutoff, "->", clustering_size, "   ", grain
00039         current_cutoff += grain
00040     return current_cutoff - grain
