00022     for traj_id in decomposed_cluster:
00023         cluster = Cluster(None, decomposed_cluster[traj_id])
00024         centers.append(cluster.calculate_medoid(matrix))
00025     center_distances = []
00026     for i in range(len(centers)-1):
00027         for j in range(i+1, len(centers)):
00028             center_distances.append(matrix[centers[i],centers[j]])

00094     ##
00095     # 
00096     #         Converts the clusters given as input into "decomposed clusters",i.e. a new data structure that divides
00097     #         the elements of the cluster according to its source trajectory.
00098     #         Example:
00099     #         cluster 1 has the following elements:
00100     #         [1, 3, 4, 7, 8, 12]
00101     #         and traj_ranges = {"traj_A":(0,6),"traj_B":(7,15)
00102     #         The resulting "decomposed cluster" will be:
00103     #         dc = {"traj_A":[1, 3, 4], "traj_B":[7, 8, 12]}
00104     # 
00105     #         @param clusters: A list containing cluster objects.
00106     #         @param traj_ranges: A dictionary that contains the starting and ending frame of each trajectory (indexed by
00107     #         a trajectory id). The numbering is accumulative, so if we have 2 trajectories of 10 models each, first will
00108     #         start in 0 and end in 9 and second will start in 10 and end in 19.
00109     # 
00110     #         @return: The list of decomposed clusterings.
00111     #         
00112     def decompose(cls, clusters, traj_ranges):
00113         set_ranges = {}
00114         for traj_id in traj_ranges:
00115             start,end = traj_ranges[traj_id]
00116             set_ranges[traj_id] = set(range(start,end+1))
00117 
00118         decomposed_clusters = {}
00119         for c in clusters:
00120             elements_set = set(c.all_elements)
00121             decomposed_cluster = {}
00122             for traj_id in set_ranges:
00123                 intersection = elements_set.intersection(set_ranges[traj_id])
00124                 if len(intersection) > 0:
00125                     # Then cluster has elements of this trajectory
00126                     decomposed_cluster[traj_id] = list(intersection)
00127             decomposed_clusters[c.id] = decomposed_cluster
00128         return decomposed_clusters
