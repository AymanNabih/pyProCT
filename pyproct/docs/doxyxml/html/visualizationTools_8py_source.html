00076 #     Calculates the bounding box that encloses all the atoms of two arrays.
00077 #     @param coordinates: An array containing coordinates for a set of conformations (3D array where dim 0 is the
00078 #     conformation, dim 1 is the atom and dim 2 the coordinate)
00079 #     @param backbone_trace: If used (and if it has length > 0) is a 2D array where dim 0 is the atom and dim 1 the
00080 #     coordinate.
00081 #     @return: A 3-tuple containing:
00082 #     - The bounding box corners.
00083 #     - The center of the b.b. .
00084 #     - The corner with maximum coordinates values.
00085 #     
00086 def calculate_bounding_box(coordinates, backbone_trace = []):
00087     coords = numpy.array(coordinates)
00088     if(len(backbone_trace)>0):
00089         [max_x,max_y,max_z] = numpy.max([numpy.max(numpy.max(coords,1),0).tolist()]+[numpy.max(backbone_trace,0).tolist()],0)
00090         [min_x,min_y,min_z] = numpy.min([numpy.min(numpy.min(coords,1),0).tolist()]+[numpy.min(backbone_trace,0).tolist()],0)
00091     else:
00092         [max_x,max_y,max_z] = numpy.max(numpy.max(coords,1),0)
00093         [min_x,min_y,min_z] = numpy.min(numpy.min(coords,1),0)
00094 
00095     center = numpy.array([min_x,min_y,min_z]) + ((numpy.array([max_x,max_y,max_z])-numpy.array([min_x,min_y,min_z])) /2.)
00096     return ([[max_x, max_y, max_z],
00097             [max_x, max_y, min_z],
00098             [max_x, min_y, max_z],
00099             [max_x, min_y, min_z],
00100             [min_x, max_y, max_z],
00101             [min_x, max_y, min_z],
00102             [min_x, min_y, max_z],
00103             [min_x, min_y, min_z]], center.tolist(), [max_x,max_y,max_z])
